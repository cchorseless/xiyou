---创建一个传送门
---@param vPortalPosition Vector 传送门的位置
---@param vTargetPosition Vector 目的地位置
---@param vForward Vector 传送门的朝向
---@return CDOTA_BaseNPC 传送门单位
function public:CreatePortal(vPortalPosition, vTargetPosition, vForward, sPortalName, bHasArrow)
	bHasArrow = bHasArrow or false
	local hPortal = CreateUnitByNameWithNewData("npc_portal", vPortalPosition, true, nil, nil, DOTA_TEAM_GOODGUYS, { UnitLabel = sPortalName or "" })
	hPortal:SetForwardVector(vForward)
	local sPosition = vTargetPosition.x .. "," .. vTargetPosition.y .. "," .. vTargetPosition.z
	local hModifier = hPortal:AddNewModifier(hPortal, nil, "modifier_portal", { vPosition = sPosition, iHasArrow = bHasArrow and 1 or 0 })
	hModifier.callback = self.OnPortalTrigger
	hModifier.context = self
	return hPortal
end
