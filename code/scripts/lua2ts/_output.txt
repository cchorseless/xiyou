// -创建一个传送门
// -@param vPortalPosition Vector 传送门的位置
// -@param vTargetPosition Vector 目的地位置
// -@param vForward Vector 传送门的朝向
// -@return CDOTA_BaseNPC 传送门单位
 CreatePortal(vPortalPosition, vTargetPosition, vForward, sPortalName, bHasArrow) {
	bHasArrow = bHasArrow || false
	let hPortal = CreateUnitByNameWithNewData("npc_portal", vPortalPosition, true, null, null, DOTA_TEAM_GOODGUYS, { UnitLabel : sPortalName || "" })
	hPortal.SetForwardVector(vForward)
	let sPosition = vTargetPosition.x + "," + vTargetPosition.y + "," + vTargetPosition.z
	let hModifier = hPortal.AddNewModifier(hPortal, null, "modifier_portal", { vPosition = sPosition, iHasArrow = bHasArrow && 1 || 0 })
	hModifier.callback = this.OnPortalTrigger
	hModifier.context = this
	return hPortal
}
